void solve()
{
    int n;
    cin >> n;
    vector<int> a(n + 1);

        cin >> a[i];

    else
    {
        cout << "YES" << endl;
        int l = 2, r = n;
        if (a[2] == a[n])

                cout << 'B';
            else
                cout << 'R';
        cout << endl;
    }
}

{
    int n;
    cin >> n;
    vector<ll> a(n + 1);
    vector<vector<ll>> f(n + 1, vector<ll>(2, 0));
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        f[i][0] = 1e18, f[i][1] = -1e18;

    #include <bits/stdc++.h>
#define endl '\n'
#define x first
#define y second
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
using db = long double;
const int N = 4e6 + 100;
const db pi = acos(-1);
const ll mod = 998244353;
using pll = pair<ll, ll>;
void solve()
{
    int n;
    cin >> n;
    vector<ll> a(n + 1);
    vector<vector<pll>> f(n + 1, vector<pll>(2, pll(0, 0)));
    for (int i = 1; i <= n; i++)
    {
        cin >> a[i];
        f[i][0] = {1e18, 0}, f[i][1] = {-1e18, 0};
    }
    if (a[1] >= 0)
    {
        f[1][0] = {a[1], 0};
        f[1][1] = {abs(a[1]), 2};
    }
    else
    {
        f[1][0] = {a[1], 1};
        f[1][1] = {abs(a[1]), 1};
    }
    for (int i = 2; i <= n; i++)
    {

        // min
        // dp val
        f[i][0].x = min(f[i][0].x, f[i - 1][0].x + a[i]);
        // dp way
        if (f[i - 1][0].x + a[i] == f[i][0].x)
            f[i][0].y = (f[i][0].y + f[i - 1][0].y) % mod;
        if (abs(f[i - 1][0].x + a[i]) == f[i][0].x)
            f[i][0].y = (f[i][0].y + f[i - 1][0].y) % mod;

        if (f[i - 1][1].x + a[i] == f[i][0].x)
            f[i][0].y = (f[i][0].y + f[i - 1][1].y) % mod;

        if (abs(f[i - 1][1].x + a[i]) == f[i][0].x)
            f[i][0].y = (f[i][0].y + f[i - 1][1].y) % mod;

        // max
        // dp val
        f[i][1].x = max({f[i][1].x, f[i - 1][1].x + a[i], abs(f[i - 1][0].x + a[i])});
        // dp way
        if (f[i - 1][0].x + a[i] == f[i][1].x)
            f[i][1].y = (f[i][1].y + f[i - 1][0].y) % mod;

        if (abs(f[i - 1][0].x + a[i]) == f[i][1].x)
            f[i][1].y = (f[i][1].y + f[i - 1][0].y) % mod;

        if (f[i - 1][1].x + a[i] == f[i][1].x)
            f[i][1].y = (f[i][1].y + f[i - 1][1].y) % mod;

        if (abs(f[i - 1][1].x + a[i]) == f[i][1].x)
            f[i][1].y = (f[i][1].y + f[i - 1][1].y) % mod;

        if (f[i][0].x == f[i][1].x)
            f[i][0].y = 0;
    }
    cout << f[n][1].y << endl;
}
int main()
{
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    // cout << setprecision(10) << endl;
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int _ = 1;
    cin >> _;
    while (_--)
        solve();
    return 0;
}
