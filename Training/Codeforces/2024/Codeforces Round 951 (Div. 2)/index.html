void solve() {
    int n;
    cin >> n;
    vector<int> a(n + 1);
    for (int i = 1; i <= n; i++)
        cin >> a[i];
    int ans = INT_MAX;
    for (int i = 2; i <= n; i++)

}

    B.XOR Sequences

    要使得\(i \oplus x=j\oplus y\),就一定要满足,相同的位置,要么同时改变,要么同时不变

    一个显然的解是 \(i=y,j=x\)

    那么,将后缀所有相同位都置0,显然也是可以的

    所以答案就是二进制位中相同后缀的长度

    C.Earning on Bets

    设总投资为\(S\),则要求转化为

    \(\sum \frac{S}{k_i} < S\)

    约掉\(S\)

    \(\sum \frac{1}{k_i} < 1\)

    用lcm通分check即可

    #include <bits/stdc++.h>
#define endl '\n'
using namespace std;
using ll = long long;
const ll mod = 998244353;
void solve() {
    ll n;
    cin >> n;
    vector<ll> k(n + 1), ans(n + 1, 0);
    for (int i = 1; i <= n; i++)
        cin >> k[i];
    ll res = 1;
    for (int i = 1; i <= n; i++)
        res = lcm(res, k[i]);
    ll sum = 0;
    for (int i = 1; i <= n; i++)
        ans[i] = res / k[i], sum += ans[i];
    if (sum >= res)
        cout << "-1" << endl;
    else {
        for (int i = 1; i <= n; i++)
            cout << ans[i] << " ";
        cout << endl;
    }
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int _ = 1;
    cin >> _;
    while (_--)
        solve();
    return 0;
}
