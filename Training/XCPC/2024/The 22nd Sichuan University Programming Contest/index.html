    int n;

    vector<pii> ans;
    for (int i = 1; i <= n; i++) {
        int u = 2 * i - 1, v = 2 * i;

        if (notpr[u + v]) {
            for (int j = (int)ans.size() - 1; j >= 0; j--) {
                if (!notpr[ans[j].fi + u] && !notpr[ans[j].se + v]) {

                    swap(ans[j].se, v);
                    break;
                }
            }
        }
        ans.push_back({u, v});
    }
    for (auto [x, y] : ans)

}

void solve() {
    // Point a(0, 0, 0), b(0, 0, 0);
    db x, r;
    cin >> r >> x;
    db ang1 = x / r;       // 第一次走的角度(弧度制)
    db r1 = r * cos(ang1); // 新平面的半径
    db l = 2 * pi * r1;

    int L = 0, R = 1e9;
    while (L < R) {
        int mid = (L + R + 1) >> 1;
        if (l * mid <= x)
            L = mid;
        else
            R = mid - 1;
    }
    db mod = x - l * L;
    db ang2 = mod / r1;
    db nlen = r * ang2;
    nlen -= x;
    cout << min(nlen, 2 * pi * r - nlen) << endl;
}

using pii = pair<int, int>;
using db = long double;
const int N = 4e6 + 100;
const db pi = acos(-1);
const ll mod = 1e9 + 7;
void solve() {
    map<ll, ll> mp;
    auto qmi = [&](ll a, ll b) -> ll {
        ll res = 1;
        while (b) {
            if (b & 1)
                res = res * a % mod;
            b >>= 1;
            a = a * a % mod;
        }
        return res;
    };
    int n, q;
    cin >> n >> q;
    vector<ll> a(n + 1, 0);
    for (int i = 1; i <= n; i++) {
        string s;
        cin >> s;
        for (auto &c : s)
            if (c <= '9' && c >= '0')
                a[i] = a[i] * 10 + c - '0';
        mp[a[i]]++;
    }

    sort(a.begin() + 1, a.end());

    for (int i = 1; i <= q; i++) {
        string s;
        cin >> s;
        ll tar = 0;
        for (auto &c : s)
            if (c <= '9' && c >= '0')
                tar = tar * 10 + c - '0';

        ll pos = lower_bound(a.begin() + 1, a.end(), tar) - a.begin();
        ll num = n - pos + 1;
        ll big = num - mp[tar];

        ll fnum = (qmi(2, num) - 1 + mod) % mod;
        ll fs = (qmi(2, pos - 1) - 1 + mod) % mod;
        ll ans = fnum * fs % mod; // 大+小

        ll equ = (qmi(2, mp[tar]) - 1 + mod) % mod;
        ll oth = qmi(2, big);
        ans = (ans + (equ * oth) % mod) % mod;
        cout << ans << endl;
    }
}

    auto len = [&](int x) -> int {
        int res = 0;
        while (x)
            x /= 10, res++;
        return res + 2;
    };
    auto print = [&](int x, int k) -> void {

                ans[i - 1]--, ans[i] += 2;

                return;
            }
        }
    }
    if (k) {
        cout << "-1" << endl;
        return;
    }
    for (int i = 0; i < n; i++)
        if (len(1 << i) + ans[i] > m) {
            cout << "-1" << endl;
            return;
        }

    cin >> n;
    if (n == 1 || n == 3 || n == 4) {
        cout << "Alice" << endl;
        return;
    }
    if (n == 2) {
        cout << "Bob" << endl;
        return;
    }
    cout << ((n % 2 == 0) ? "Alice" : "Bob") << endl;
}
int main() {
#ifndef ONLINE_JUDGE
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int _ = 1;
    cin >> _;
    while (_--)
        solve();
    return 0;
}
