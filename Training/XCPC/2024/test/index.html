const ll mod = 998244353;
int f[2][N][N * 2]; // i j 0
void solve() {
    int n, m, p, q, u, to;
    cin >> n >> m >> p >> q;
    vector<vector<int>> a(n + 1, vector<int>(m + 1));
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            cin >> a[i][j];
    if (!a[1][1])
        f[1][1][1] = 1;
    else
        f[1][1][0] = 1;
    for (int i = 1; i <= n; i++) {
        u = i % 2, to = (i + 1) % 2;
        for (int j = 1; j <= m; j++) // clear
            for (int k = 0; k <= n + m - 1; k++)
                f[to][j][k] = 0;

        for (int j = 1; j <= m; j++)
            for (int k = 0; k <= n + m - 1; k++) {
                if (i + 1 <= n)
                    f[to][j][k + !a[i + 1][j]] = (f[to][j][k + !a[i + 1][j]] + f[u][j][k]) % mod;
                if (j + 1 <= m)
                    f[u][j + 1][k + !a[i][j + 1]] = (f[u][j + 1][k + !a[i][j + 1]] + f[u][j][k]) % mod;
            }
    }
    ll ans = 0;
    for (int i = 0; i <= n + m - 1; i++)
        if (i >= p && n + m - 1 - i >= q) {
            ans = (ans + f[n % 2][m][i]) % mod;
        }
    cout << ans << endl;
}

        solve();
