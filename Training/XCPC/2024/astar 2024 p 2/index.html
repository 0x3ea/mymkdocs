    #include <bits/stdc++.h>
#define endl '\n'
using namespace std;
using ll = long long;
struct segtree {
    struct node {
        ll t, val, sz, mi;
    };
    int n;
    vector<node> a;
    segtree(int _n) : n(_n * 4 + 10), a(n + 10) {}
    void update(int id) {
        a[id].val = a[id * 2].val + a[id * 2 + 1].val;
        a[id].mi = min(a[id * 2].mi, a[id * 2 + 1].mi);
    }
    void settag(int id, int t) {
        a[id].val = a[id].val + t * a[id].sz;
        a[id].t = a[id].t + t;
        a[id].mi = a[id].mi + t;
    }
    void push_down(int id) {
        if (a[id].t) {
            settag(id * 2, a[id].t);
            settag(id * 2 + 1, a[id].t);
            a[id].t = 0;
        }
    }
    void build(int id, int l, int r, vector<int> &arr) {
        a[id].t = 0, a[id].sz = r - l + 1;
        if (l == r)
            a[id].val = a[id].mi = arr[l];
        else {
            int mid = l + r >> 1;
            build(id * 2, l, mid, arr);
            build(id * 2 + 1, mid + 1, r, arr);
            update(id);
        }
    }
    void modify(int id, int l, int r, int ql, int qr, int t) {
        if (r < ql || l > qr)
            return;
        if (l >= ql && r <= qr) {
            settag(id, t);
            return;
        }
        int mid = l + r >> 1;
        push_down(id);
        if (ql <= mid)
            modify(id * 2, l, mid, ql, qr, t);
        if (qr > mid)
            modify(id * 2 + 1, mid + 1, r, ql, qr, t);
        update(id);
    }
    ll query(int id, int l, int r, int ql, int qr) {
        if (r < ql || l > qr)
            return 0ll;
        if (l >= ql && r <= qr)
            return a[id].mi;
        int mid = l + r >> 1;
        push_down(id);
        ll res = 1e18;
        if (ql <= mid)
            res = min(res, query(id * 2, l, mid, ql, qr));
        if (qr > mid)
            res = min(res, query(id * 2 + 1, mid + 1, r, ql, qr));
        return res;
    }
};

void solve() {
    string s;
    cin >> s;
    int n = s.size();
    s = '#' + s;
    vector<int> a(n + 10, 0);
    for (int i = 1; i <= n; i++)
        a[i] = a[i - 1] + (s[i] == '(' ? 1 : -1);
    if (a[n] == 0) {
        cout << '0' << endl;
        return;
    }
    segtree seg(n);
    seg.build(1, 1, n, a);
    int ans = 0;
    for (int i = 1; i <= n; i++) {
        if (s[i] == '(') {
            seg.modify(1, 1, n, i, n, -2);
            if (seg.query(1, 1, n, n, n) == 0 && seg.query(1, 1, n, 1, n) >= 0)
                ans++;
            seg.modify(1, 1, n, i, n, 2);
        } else {
            seg.modify(1, 1, n, i, n, 2);
            if (seg.query(1, 1, n, n, n) == 0 && seg.query(1, 1, n, 1, n) >= 0)
                ans++;
            seg.modify(1, 1, n, i, n, -2);
        }
    }
    cout << ans << endl;
}
int main() {
#ifdef ix3ea
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int _ = 1;
    //  cin >> _;
    while (_--)
        solve();
    return 0;
}

    #include <bits/stdc++.h>
#define endl '\n'
#define x first
#define y second
using namespace std;
using ll = long long;
using pll = pair<ll, ll>;
struct segtree {
    struct node {
        ll t, val1, val2, sz;
    };
    int n;
    vector<node> a;
    segtree(int _n) : n(_n * 4 + 10), a(n + 10) {}
    void update(int id) {
        a[id].val1 = a[id * 2].val1 + a[id * 2 + 1].val1;
        a[id].val2 = a[id * 2].val2 + a[id * 2 + 1].val2;
    }
    void settag(int id, int t) {
        a[id].val1 = a[id].sz - a[id].val1;
        a[id].val2 = a[id].sz - a[id].val2;
        a[id].t = (a[id].t + t) % 2;
    }
    void push_down(int id) {
        if (a[id].t) {
            settag(id * 2, a[id].t);
            settag(id * 2 + 1, a[id].t);
            a[id].t = 0;
        }
    }
    void build(int id, int l, int r, vector<int> &arr) {
        a[id].t = 0, a[id].sz = r - l + 1;
        if (l == r) {
            a[id].val1 = arr[l];
            a[id].val2 = 1 - arr[l];
        } else {
            int mid = l + r >> 1;
            build(id * 2, l, mid, arr);
            build(id * 2 + 1, mid + 1, r, arr);
            update(id);
        }
    }
    void modify(int id, int l, int r, int ql, int qr, int t) {
        if (r < ql || l > qr)
            return;
        if (l >= ql && r <= qr) {
            settag(id, t);
            return;
        }
        int mid = l + r >> 1;
        push_down(id);
        if (ql <= mid)
            modify(id * 2, l, mid, ql, qr, t);
        if (qr > mid)
            modify(id * 2 + 1, mid + 1, r, ql, qr, t);
        update(id);
    }
    pll query(int id, int l, int r, int ql, int qr) {
        if (r < ql || l > qr)
            return {0ll, 0ll};
        if (l >= ql && r <= qr)
            return {a[id].val1, a[id].val2};
        int mid = l + r >> 1;
        push_down(id);
        pll res = {0ll, 0ll};
        if (ql <= mid) {
            auto tmp = query(id * 2, l, mid, ql, qr);
            res.x += tmp.x, res.y += tmp.y;
        }
        if (qr > mid) {
            auto tmp = query(id * 2 + 1, mid + 1, r, ql, qr);
            res.x += tmp.x, res.y += tmp.y;
        }
        return res;
    }
};

void solve() {
    int n, q;
    cin >> n >> q;
    vector<int> s(n + 10, 0);
    for (int i = 1; i <= n; i++) {
        char c;
        cin >> c;
        if (c - '0' == i % 2)
            s[i] = 1;
    }
    segtree seg(n);
    seg.build(1, 1, n, s);
    for (int i = 1; i <= q; i++) {
        int t, l, r;
        cin >> t >> l >> r;
        if (t == 1)
            seg.modify(1, 1, n, l, r, 1);
        else {
            auto res = seg.query(1, 1, n, l, r);
            cout << min(res.x, res.y) << endl;
        }
    }
}
int main() {
#ifdef ix3ea
    freopen("in.txt", "r", stdin);
    freopen("out.txt", "w", stdout);
#endif
    ios::sync_with_stdio(false);
    cin.tie(0), cout.tie(0);
    int _ = 1;
    //  cin >> _;
    while (_--)
        solve();
    return 0;
}
